Quick Start Examples

Get started with our API in minutes

Python

import requests

# Get market data
response = requests.get(
    'https://api.wizdata.com/market-data',
    params={
        'symbol': 'JSE:AGL',
        'timeframe': '1d',
        'limit': 100
    },
    headers={
        'Authorization': 'Bearer YOUR_API_KEY'
    }
)

data = response.json()
print(f"Latest price: {data['close']}")

JavaScript

// WebSocket real-time data
const ws = new WebSocket('wss://api.wizdata.com/ws');

ws.onopen = () => {
    ws.send(JSON.stringify({
        action: 'subscribe',
        symbols: ['JSE:AGL', 'NSE:DANGCEM'],
        type: 'real-time-quotes'
    }));
};

ws.onmessage = (event) => {
    const data = JSON.parse(event.data);
    console.log('Price update:', data);
};
